// 3*2
// arr1 = [[1, 4],
//         [3, 2],
//         [4, 1]];
// 2*2
// arr2 = [[3, 3],
//         [3, 2]]

//[1, 4]
// 1 * 3 3   1 * 3 3
// 4 * 3 12  4 * 2 8
// [15, 11]

//[3, 2]
// 3 * 3 9   3 * 3 9
// 2 * 3 6   2 * 2 4
// [15, 13]

//[4, 1]
// 4 * 3 12  4 * 3 12
// 1 * 3 3   1 * 2 2
// [15, 14]
function solution(arr1, arr2) {
  let answer = new Array()
  for (let k = 0; k < arr1.length; k++) {
    answer[k] = new Array()
    for (let i = 0; i < arr2[0].length; i++) {
      let sum = 0
      for (let j = 0; j < arr1[0].length; j++) {
        sum += arr1[k][j] * arr2[j][i]
      }
      answer[k].push(sum)
    }
  }
  return answer
}
